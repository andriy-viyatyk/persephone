## Implementation Plan

### 2.1. Define IEditorRegistry interface

**File:** `/src/renderer/api/types.ts` (add to existing)

```typescript
export interface IEditorInfo {
  /** Editor ID (e.g., "monaco", "grid-json", "pdf-view") */
  readonly id: string;
  /** Display name (e.g., "Text Editor", "Grid", "PDF") */
  readonly name: string;
  /** Category: "page-editor" (standalone) or "content-view" (text-based) */
  readonly category: string;
}

export interface IEditorRegistry {
  /** Get all registered editors */
  getAll(): IEditorInfo[];

  /** Get editor info by ID. Returns undefined for unknown IDs. */
  getById(id: string): IEditorInfo | undefined;

  /** Resolve the best editor for a file path. Returns undefined if no match. */
  resolve(filePath: string): IEditorInfo | undefined;

  /** Resolve just the editor ID for a file path */
  resolveId(filePath: string): string | undefined;

  /** Get available view switch options for a language */
  getSwitchOptions(languageId: string, filePath?: string): {
    options: string[];
    getOptionLabel: (option: string) => string;
  };
}
```

**Notes:**
- `IEditorInfo` is a read-only subset of `EditorDefinition` — hides `loadModule()`, `acceptFile()`, and other internal details
- `register()` is NOT exposed — editors are registered at startup, not by scripts
- No events needed — editor list is static after startup

### 2.2. Implement EditorRegistryApi wrapper

**File:** `/src/renderer/api/editors.ts`

```typescript
import { editorRegistry } from "../editors/registry";
import type { IEditorRegistry, IEditorInfo } from "./types";

class EditorRegistryApi implements IEditorRegistry {
  getAll(): IEditorInfo[] {
    return editorRegistry.getAll().map(toEditorInfo);
  }

  getById(id: string): IEditorInfo | undefined {
    const def = editorRegistry.getById(id as any);
    return def ? toEditorInfo(def) : undefined;
  }

  resolve(filePath: string): IEditorInfo | undefined {
    const def = editorRegistry.resolve(filePath);
    return def ? toEditorInfo(def) : undefined;
  }

  resolveId(filePath: string): string | undefined {
    return editorRegistry.resolveId(filePath);
  }

  getSwitchOptions(languageId: string, filePath?: string) {
    return editorRegistry.getSwitchOptions(languageId, filePath);
  }
}

function toEditorInfo(def: any): IEditorInfo {
  return { id: def.id, name: def.name, category: def.category };
}

export const editors = new EditorRegistryApi();
```
